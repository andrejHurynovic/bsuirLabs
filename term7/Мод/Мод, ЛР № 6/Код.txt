  	GENERATE  100,20  ;генерация транзактов с интервалом 100 +/-20 усл. ед.
  	QUEUE     pribor  ;транзакт становится в очередь
	SEIZE     pribor  ;транзакт занимает прибор
  	DEPART  pribor  ;транзакт покидает очередь, важно, что он освобождает очередь только после занятия прибора
  	ADVANCE   50,20  ;прибор обрабатывает транзакт с продолжительностью 50 +/-20 усл. ед.
 	RANSFER   100,EXIT1,MODIF1  ;с вероятностью 0.1 транзакт переходит на метку MODIF1


MODIF1  ADVANCE   20,5  ;прибор дополнительно обрабатывает транзакт с продолжительностью 20 +/-5 усл. 
	TRANSFER  ,EXIT1  ;транзакт переходит на метку EXIT1
      
      
EXIT1  	RELEASE   pribor  ;транзакт освобождает прибор
  	TERMINATE 1  ;транзакт выходит из системы
